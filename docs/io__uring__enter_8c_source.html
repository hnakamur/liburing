138 void
139 reap_events(struct io_uring *ring, unsigned nr)
140 {
141  int ret;
142  unsigned left = nr;
143  struct io_uring_cqe *cqe;
144  struct iovec *iov;
145  struct timeval start, now, elapsed;
146 
147  printf("Reaping %u I/Os\n", nr);
148  gettimeofday(&start, NULL);
149  while (left) {
150  ret = io_uring_wait_cqe(ring, &cqe);
151  if (ret < 0) {
152  printf("io_uring_wait_cqe returned %d\n", ret);
153  printf("expected success\n");
154  exit(1);
155  }
156  if (cqe->res != 4096)
157  printf("cqe->res: %d, expected 4096\n", cqe->res);
158  iov = io_uring_cqe_get_data(cqe);
159  free(iov->iov_base);
160  free(iov);
161  left--;
162  io_uring_cqe_seen(ring, cqe);
163 
164  gettimeofday(&now, NULL);
165  timersub(&now, &start, &elapsed);
166  if (elapsed.tv_sec > 10) {
167  printf("Timed out waiting for I/Os to complete.\n");
168  printf("%u expected, %u completed\n", nr, left);
169  break;
170  }
171  }

206 {

208  unsigned int status = 0;
209  struct io_uring ring;
210  struct io_uring_sq *sq = &ring.sq;
211  unsigned ktail, mask, index;
212  unsigned sq_entries;
213  unsigned completed, dropped;
214 
215  ret = io_uring_queue_init(IORING_MAX_ENTRIES, &ring, 0);
216  if (ret < 0) {
217  perror("io_uring_queue_init");
218  exit(1);
219  }
220  mask = *sq->kring_mask;
221 
222  /* invalid flags */
223  status |= try_io_uring_enter(ring.ring_fd, 1, 0, ~0U, NULL, -1, EINVAL);
224 
225  /* invalid fd, EBADF */
226  status |= try_io_uring_enter(-1, 0, 0, 0, NULL, -1, EBADF);
227 
228  /* valid, non-ring fd, EOPNOTSUPP */
229  status |= try_io_uring_enter(0, 0, 0, 0, NULL, -1, EOPNOTSUPP);
230 
231  /* to_submit: 0, flags: 0; should get back 0. */
232  status |= try_io_uring_enter(ring.ring_fd, 1, 0, 0, NULL, 0, 0);
233 
234  /* fill the sq ring */
235  sq_entries = *ring.sq.kring_entries;
236  submit_io(&ring, sq_entries);
237  printf("Waiting for %u events\n", sq_entries);
238  ret = __sys_io_uring_enter(ring.ring_fd, 0, sq_entries,
239  IORING_ENTER_GETEVENTS, NULL);
240  if (ret < 0) {
241  perror("io_uring_enter");
242  status = 1;
243  } else {
244  /*
245  * This is a non-IOPOLL ring, which means that io_uring_enter
246  * should not return until min_complete events are available
247  * in the completion queue.
248  */
249  completed = *ring.cq.ktail - *ring.cq.khead;
250  if (completed != sq_entries) {
251  printf("Submitted %u I/Os, but only got %u completions\n",
252  sq_entries, completed);
253  status = 1;
254  }
255  reap_events(&ring, sq_entries);
256  }
257 
258  /*
259  * Add an invalid index to the submission queue. This should
260  * result in the dropped counter increasing.
261  */
262  printf("Submitting invalid sqe index.\n");
263  index = *sq->kring_entries + 1; // invalid index
264  dropped = *sq->kdropped;
265  ktail = *sq->ktail;
266  sq->array[ktail & mask] = index;
267  ++ktail;
268  /*
269  * Ensure that the kernel sees the SQE update before it sees the tail
270  * update.
271  */
272  io_uring_smp_store_release(sq->ktail, ktail);
273 
274  ret = __sys_io_uring_enter(ring.ring_fd, 1, 0, 0, NULL);
275  /* now check to see if our sqe was dropped */
276  if (*sq->kdropped == dropped) {
277  printf("dropped counter did not increase\n");
278  status = 1;
279  }
280 
281  if (!status) {
282  printf("PASS\n");
283  return 0;
284  }
285 
286  printf("FAIL\n");
287  return -1;
288 }
