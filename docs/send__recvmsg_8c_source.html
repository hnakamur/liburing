91 static int do_recvmsg(struct io_uring *ring, struct iovec *iov,
92  struct recv_data *rd)

98  if (rd->no_buf_add && rd->buf_select)
99  return 0;
100  fprintf(stderr, "%s: failed cqe: %d\n", __FUNCTION__, cqe->res);
101  goto err;
102  }

105  if (bid != BUF_BID)

110

113  goto err;
114  }
115 
116  if (cqe->res -1 != strlen(str)) {
117  fprintf(stderr, "got wrong length: %d/%d\n", cqe->res,
118  (int) strlen(str) + 1);
119  goto err;
120  }
121 
122  if (strcmp(str, iov->iov_base)) {
123  fprintf(stderr, "string mismatch\n");
124  goto err;
125  }
