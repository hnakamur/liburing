87 int await_delay(async_context *pctx, time_t seconds) {
88  struct itimerspec exp = {
89  .it_interval = {},
90  .it_value = { seconds, 0 },
91  };
92  int tfd = timerfd_create(CLOCK_MONOTONIC, TFD_NONBLOCK);
93  if (tfd < 0) {
94  perror("timerfd_create");
95  return -1;
96  }
97  if (timerfd_settime(tfd, 0, &exp, NULL)) {
98  perror("timerfd_settime");
99  close(tfd);
100  return -1;
101  }
102 
103  int ret = await_poll(pctx, tfd, POLLIN);
104  assert(ret == POLLIN);
105 
106  close(tfd);
107 
108  return 0;
109 }

128 static int copy_file(async_context *pctx, int infd, int outfd, struct iovec* piov)
129 {
130  off_t offset = 0;
131 
132  for (;;) {
133  ssize_t bytes_read;

136  if ((bytes_read = await_readv(pctx, infd, piov, 1, offset)) < 0) {
137  perror("await_readv");

139  }
140  if (bytes_read == 0)
141  return 0;
142

144 
145  printf("%d->%d: writev %ld bytes from %ld\n", infd, outfd, (long) piov->iov_len, (long) offset);
146  if (await_writev(pctx, outfd, piov, 1, offset) != bytes_read) {
147  perror("await_writev");
148  return 1;
149  }
150  if (bytes_read < BS)
151  return 0;
152  offset += bytes_read;
153 
154  printf("%d->%d: wait %ds\n", infd, outfd, 1);
155  await_delay(pctx, 1);
156  }

186 int main(int argc, char *argv[])
187 {
188  struct io_uring ring;
189  int i, req_count, ret;
190  int success = 0, failure = 0;
191 
192  if (argc < 3) {
193  fprintf(stderr, "%s: infile1 outfile1 [infile2 outfile2 [...]]\n", argv[0]);
194  return 1;
195  }
196 
197  ret = io_uring_queue_init(QD, &ring, 0);
198  if (ret < 0) {
199  fprintf(stderr, "queue_init: %s\n", strerror(-ret));
200  return -1;
201  }
202 
203  req_count = (argc - 1) / 2;
204  printf("copying %d files...\n", req_count);
205 
206  for (i = 1; i < argc; i += 2) {
207  int infd, outfd;
208 
209  async_context *pctx = malloc(sizeof(*pctx));
210 
211  if (!pctx || setup_context(pctx, &ring))
212  return 1;
213 
214  infd = open(argv[i], O_RDONLY);
215  if (infd < 0) {
216  perror("open infile");
217  return 1;
218  }
219  outfd = open(argv[i + 1], O_WRONLY | O_CREAT | O_TRUNC, 0644);
220  if (outfd < 0) {
221  perror("open outfile");
222  return 1;
223  }
224 
225  arguments_bundle *pbundle = malloc(sizeof(*pbundle));
226  pbundle->pctx = pctx;
227  pbundle->psuccess = &success;
228  pbundle->pfailure = &failure;
229  pbundle->infd = infd;
230  pbundle->outfd = outfd;
231 
232  makecontext(&pctx->ctx_fnew, (void (*)(void)) copy_file_wrapper, 1, pbundle);
233 
234  if (swapcontext(&pctx->ctx_main, &pctx->ctx_fnew)) {
235  perror("swapcontext");
236  return 1;
237  }
238  }
239 
240  /* event loop */
241  while (success + failure < req_count) {
242  struct io_uring_cqe *cqe;
243 
244  /* usually be timed waiting */
245  ret = io_uring_submit_and_wait(&ring, 1);
246  if (ret < 0) {
247  fprintf(stderr, "submit_and_wait: %s\n", strerror(-ret));
248  return 1;
249  }
250 
251  ret = io_uring_wait_cqe(&ring, &cqe);
252  if (ret < 0) {
253  fprintf(stderr, "wait_cqe: %s\n", strerror(-ret));
254  return 1;
255  }
256 
257  async_context *pctx = io_uring_cqe_get_data(cqe);
258 
259  if (swapcontext(&pctx->ctx_main, &pctx->ctx_fnew)) {
260  perror("swapcontext");
261  return 1;
262  }
263  }
264 
265  io_uring_queue_exit(&ring);
266 
267  printf("finished with %d success(es) and %d failure(s)\n", success, failure);
268 
269  return failure > 0;
270 }
