83  * For users that want to specify sq_thread_cpu or sq_thread_idle, this
84  * interface is a convenient helper for mmap()ing the rings.
85  * Returns -1 on error, or zero on success. On success, 'ring'

88 int io_uring_queue_mmap(int fd, struct io_uring_params *p, struct io_uring *ring)

101 /*
102  * Ensure that the mmap'ed rings aren't available to a child after a fork(2).
103  * This uses madvise(..., MADV_DONTFORK) on the mmap'ed ranges.
104  */
105 int io_uring_ring_dontfork(struct io_uring *ring)
106 {
107  size_t len;
108  int ret;
109 
110  if (!ring->sq.ring_ptr || !ring->sq.sqes || !ring->cq.ring_ptr)
111  return -EINVAL;
112 
113  len = *ring->sq.kring_entries * sizeof(struct io_uring_sqe);
114  ret = madvise(ring->sq.sqes, len, MADV_DONTFORK);
115  if (ret == -1)
116  return -errno;
117 
118  len = ring->sq.ring_sz;
119  ret = madvise(ring->sq.ring_ptr, len, MADV_DONTFORK);
120  if (ret == -1)
121  return -errno;
122 
123  if (ring->cq.ring_ptr != ring->sq.ring_ptr) {
124  len = ring->cq.ring_sz;
125  ret = madvise(ring->cq.ring_ptr, len, MADV_DONTFORK);
126  if (ret == -1)
127  return -errno;
128  }
129 
130  return 0;
131 }
