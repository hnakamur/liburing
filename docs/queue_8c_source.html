17 /*
18  * Returns true if we're not using SQ thread (thus nobody submits but us)
19  * or if IORING_SQ_NEED_WAKEUP is set, so submit thread must be explicitly
20  * awakened. For the latter case, we set the thread wakeup flag.
21  */
22 static inline bool sq_ring_needs_enter(struct io_uring *ring,
23  unsigned submitted, unsigned *flags)
24 {
25  if (!(ring->flags & IORING_SETUP_SQPOLL) && submitted)
26  return true;
27  if (IO_URING_READ_ONCE(*ring->sq.kflags) & IORING_SQ_NEED_WAKEUP) {

29  return true;
30  }
31 
32  return false;
33 }

35 int __io_uring_get_cqe(struct io_uring *ring, struct io_uring_cqe **cqe_ptr,
36  unsigned submit, unsigned wait_nr, sigset_t *sigmask)
37 {
38  struct io_uring_cqe *cqe = NULL;
39  const int to_wait = wait_nr;
40  int ret = 0, err;
41 
42  do {
43  unsigned flags = 0;
44 
45  err = __io_uring_peek_cqe(ring, &cqe);
46  if (err)
47  break;
48  if (!cqe && !to_wait && !submit) {
49  err = -EAGAIN;
50  break;
51  }
52  if (wait_nr)
53  flags = IORING_ENTER_GETEVENTS;
54  if (submit)
55  sq_ring_needs_enter(ring, submit, &flags);
56  if (wait_nr || submit)
57  ret = __sys_io_uring_enter(ring->ring_fd, submit,
58  wait_nr, flags, sigmask);
59  if (ret < 0) {
60  err = -errno;
61  } else if (ret == submit) {
62  submit = 0;
63  wait_nr = 0;
64  } else {
65  submit -= ret;
66  }
67  if (cqe)
68  break;
69  } while (!err);
70 
71  *cqe_ptr = cqe;
72  return err;
73 }

102 /*
103  * Sync internal state with kernel ring state on the SQ side. Returns the
104  * number of pending items in the SQ ring, for the shared ring.
105  */
106 static int __io_uring_flush_sq(struct io_uring *ring)
107 {
108  struct io_uring_sq *sq = &ring->sq;
109  const unsigned mask = *sq->kring_mask;
110  unsigned ktail, to_submit;
111 
112  if (sq->sqe_head == sq->sqe_tail) {
113  ktail = *sq->ktail;
114  goto out;
115  }
116 
117  /*
118  * Fill in sqes that we have queued up, adding them to the kernel ring
119  */
120  ktail = *sq->ktail;
121  to_submit = sq->sqe_tail - sq->sqe_head;
122  while (to_submit--) {
123  sq->array[ktail & mask] = sq->sqe_head & mask;
124  ktail++;
125  sq->sqe_head++;
126  }
127 
128  /*
129  * Ensure that the kernel sees the SQE updates before it sees the tail
130  * update.
131  */
132  io_uring_smp_store_release(sq->ktail, ktail);
133 out:
134  return ktail - *sq->khead;
135 }

137 /*
138  * Like io_uring_wait_cqe(), except it accepts a timeout value as well. Note
139  * that an sqe is used internally to handle the timeout. Applications using
140  * this function must never set sqe->user_data to LIBURING_UDATA_TIMEOUT!
141  *
142  * If 'ts' is specified, the application need not call io_uring_submit() before
143  * calling this function, as we will do that on its behalf. From this it also
144  * follows that this function isn't safe to use for applications that split SQ
145  * and CQ handling between two threads and expect that to work without
146  * synchronization, as this function manipulates both the SQ and CQ side.
147  */
148 int io_uring_wait_cqes(struct io_uring *ring, struct io_uring_cqe **cqe_ptr,
149  unsigned wait_nr, struct __kernel_timespec *ts,
150  sigset_t *sigmask)
151 {
152  unsigned to_submit = 0;
153 
154  if (ts) {
155  struct io_uring_sqe *sqe;
156  int ret;
157 
158  /*
159  * If the SQ ring is full, we may need to submit IO first
160  */
161  sqe = io_uring_get_sqe(ring);
162  if (!sqe) {
163  ret = io_uring_submit(ring);
164  if (ret < 0)
165  return ret;
166  sqe = io_uring_get_sqe(ring);
167  if (!sqe)
168  return -EAGAIN;
169  }
170  io_uring_prep_timeout(sqe, ts, wait_nr, 0);
171  sqe->user_data = LIBURING_UDATA_TIMEOUT;
172  to_submit = __io_uring_flush_sq(ring);
173  }
174 
175  return __io_uring_get_cqe(ring, cqe_ptr, to_submit, wait_nr, sigmask);
176 }

194 static int __io_uring_submit(struct io_uring *ring, unsigned submitted,
195  unsigned wait_nr)
196 {
197  unsigned flags;
198  int ret;
199 
200  flags = 0;
201  if (sq_ring_needs_enter(ring, submitted, &flags) || wait_nr) {
202  if (wait_nr || (ring->flags & IORING_SETUP_IOPOLL))
203  flags |= IORING_ENTER_GETEVENTS;
204 
205  ret = __sys_io_uring_enter(ring->ring_fd, submitted, wait_nr,
206  flags, NULL);
207  if (ret < 0)

209  } else
210  ret = submitted;
